109 : __DATE__ [2] == 'b' ? 2 \
110 : __DATE__ [2] == 'r' ? (__DATE__ [0] == 'M' ? 3 : 4) \
111 : __DATE__ [2] == 'y' ? 5 \
112 : __DATE__ [2] == 'l' ? 7 \
113 : __DATE__ [2] == 'g' ? 8 \
114 : __DATE__ [2] == 'p' ? 9 \
115 : __DATE__ [2] == 't' ? 10 \
116 : __DATE__ [2] == 'v' ? 12 : 12)
117
118#define YEAR ((__DATE__ [7] - '0') * 1000 + \
119 (__DATE__ [8] - '0') * 100 + \
120 (__DATE__ [9] - '0') * 10 + \
121 (__DATE__ [10] - '0') )
122
123#define DAY ((__DATE__ [4] == ' ' ? 0 : __DATE__ [4] - '0') * 10 + (__DATE__ [5] - '0'))

125#define DATE_AS_INT ((YEAR << 16) + (MONTH << 8) + DAY )
